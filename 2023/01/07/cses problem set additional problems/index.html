1087 - shortest subsequence

code

void solve(){ string str; map<char,int> mp;

mp['A'] = 0, mp['T'] = 1, mp['C'] = 2, mp['G'] = 3;
string ch = "ATCG";

cin >> str;

int n = sz(str);
int arr[4] = {0,0,0,0};
string ans;

rep(i,0,n){
    arr[mp[str[i]]] = 1;
    bool check = 1;
    rep(i,0,4) if(arr[i]==0) check = 0;
    if(check){
        ans.push_back(str[i]);
        rep(i,0,4) arr[i] = 0;
    }
}

rep(i,0,4) if(arr[i]==0){
    ans.push_back(ch[i]);
    cout << ans << endl;
    return;
}

}

1146 - counting bits

code

void solve(){ ll n; cin >> n;

n++;
ll cnt = 0;
for(ll i=1;(1ll<<(i-1))<=n;i++){
    ll pw = (1ll<<i);
    cnt += (n/pw)*(pw/2)+max(0ll,n%pw-pw/2);
}

cout << cnt << endl;

}
